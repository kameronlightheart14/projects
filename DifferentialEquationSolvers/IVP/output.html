def euler(f,x0,t):
    """Numerically approximates the solution to the IVP:
    
    x'(t) = f(x(t),t)
    x(t0) = x0
    
    using the Euler method.
    Parameters:
        f (function): The right-hand side of the ODE
        x0 ((m,) ndarray): The initial condition
        t ((n,) ndarray): The array of time values
    Returns:
        ((n,m) ndarray): The approximate solution, where x[i] Ã¢â€°Ë† x(t_i)
    """
    # Calculate Step
    h = t[1] - t[0]
    # Grab dimensions
    m = len(x0)
    n = len(t)
    # Initialize array of approximating values
    x = np.zeros((m, n))
    x[:,0] = x0
    # Run through the algorithm to approximate values
    for i in range(1, n):
        x[:,i] = x0 + h*f(x0, t[i])
        x0 = x[:,i]
    return x

def midpoint(f,x0,t):
    """Numerically approximates the solution to the IVP:
    
    x'(t) = f(x(t),t)
    x(t0) = x0
    
    using the midpoint method.
    Parameters:
        f (function): The right-hand side of the ODE
        x0 ((m,) ndarray): The initial condition
        t ((n,) ndarray): The array of time values
    Returns:
        ((n,m) ndarray): The approximate solution, where x[i] Ã¢â€°Ë† x(t_i)
    """
    # Calculate Step
    h = t[1] - t[0]
    # Grab dimensions
    m = len(x0)
    n = len(t)
    # Initialize array of approximating values
    x = np.zeros((m, n))
    x[:,0] = x0
    # Run through the algorithm to approximate values
    for i in range(1, n):
        x_hat = x0 + (h/2)*f(x0, t[i])
        x[:,i] = x0 + h*f(x_hat, t[i] + (h/2))
        x0 = x[:,i]
    return x

def rk4(f,x0,t):
    """Numerically approximates the solution to the IVP:
    
    x'(t) = f(x(t),t)
    x(t0) = x0
    
    using a fourth-order Runge-Kutta method.
    Parameters:
        f (function): The right-hand side of the ODE
        x0 ((m,) ndarray): The initial condition
        t ((n,) ndarray): The array of time values
    Returns:
        ((n,m) ndarray): The approximate solution, where x[i] Ã¢â€°Ë† x(t_i)
    """
    # Calculate Step
    h = t[1] - t[0]
    # Grab dimensions
    m = len(x0)
    n = len(t)
    # Initialize array of approximating values
    x = np.zeros((m, n))
    x[:,0] = x0
    # Run through the algorithm to approximate values
    for i in range(1, n):
        k1 = f(x0, t[i-1])
        k2 = f(x0 + (h/2)*k1, t[i-1] + (h/2))
        k3 = f(x0 + (h/2)*k2, t[i-1] + (h/2))
        k4 = f(x0 + h*k3, t[i])
        x[:,i] = x0 + (h/6)*(k1 + 2*k2 + 2*k3 + k4)
        x0 = x[:,i]
    return x

def prob4(x, t, gamma=0.5, omega=1.5):
    x1 = x[0]
    x2 = x[1]
    return np.array([x2, (2*np.cos(omega*t)-gamma*x2-2*x1)/2])
def wrapper(x, t):
    return prob4(x, t, gamma=0.1, omega=1.1)
def wrapper2(x, t):
    return prob4(x, t, gamma=0.0, omega=1.0)
